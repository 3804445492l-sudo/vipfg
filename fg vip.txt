    local FlingState = {Active = false, Conn = nil, OriginalPos = nil}

    local function EndFling()
        FlingState.Active = false
        if FlingState.Conn then FlingState.Conn:Disconnect() end
        local Char = LocalPlayer.Character
        if Char then
            local Root = Char:FindFirstChild("HumanoidRootPart")
            local Hum = Char:FindFirstChild("Humanoid")
            if Hum then 
                Hum.PlatformStand = false 
                Hum:ChangeState(Enum.HumanoidStateType.GettingUp)
            end
            if Root then
                Root.Velocity = Vector3.new(0,0,0)
                Root.RotVelocity = Vector3.new(0,0,0)
                Root.AssemblyLinearVelocity = Vector3.new(0,0,0)
                Root.AssemblyAngularVelocity = Vector3.new(0,0,0)
                if FlingState.OriginalPos then
                    Root.CFrame = FlingState.OriginalPos
                    FlingState.OriginalPos = nil
                    Library:Notify("✅ 甩飞结束，已安全归位", 2)
                end
            end
            for _, v in pairs(Char:GetDescendants()) do
                if v:IsA("BasePart") then v.CanCollide = true end
            end
        end
        Camera.CameraSubject = LocalPlayer.Character.Humanoid
    end

    local function FlingLogic(TargetName, Mode)
        local Target = Players:FindFirstChild(TargetName)
        if not Target or not Target.Character then Library:Notify("❌ 目标无效", 2) return end
        local MyChar = LocalPlayer.Character
        if not MyChar then return end
        local MyRoot = MyChar:FindFirstChild("HumanoidRootPart")
        local MyHum = MyChar:FindFirstChild("Humanoid")
        if not MyRoot or not MyHum then return end

        if not FlingState.OriginalPos then FlingState.OriginalPos = MyRoot.CFrame end
        FlingState.Active = true
        Library:Notify("🚀 飞哥甩飞启动: " .. TargetName, 3)
        MyHum.PlatformStand = true

        FlingState.Conn = RunService.Heartbeat:Connect(function()
            if not FlingState.Active or not Target or not Target.Character or not Target.Character:FindFirstChild("HumanoidRootPart") then
                EndFling()
                return
            end
            local T_Root = Target.Character.HumanoidRootPart
            if T_Root.Velocity.Magnitude > 100 or (T_Root.Position - FlingState.OriginalPos.Position).Magnitude > 500 then
                EndFling()
                return
            end
            MyRoot.CFrame = T_Root.CFrame * CFrame.new(math.random(-2,2), math.random(-2,2), math.random(-2,2)) * CFrame.Angles(math.random(0,3), math.random(0,3), math.random(0,3))
            local Vel = Vector3.new(9e9, 9e9, 9e9)
            MyRoot.Velocity = Vel
            MyRoot.RotVelocity = Vel
            MyRoot.AssemblyLinearVelocity = Vel
            for _, v in pairs(MyChar:GetDescendants()) do
                if v:IsA("BasePart") then
                    v.CanCollide = false
                    v.AssemblyLinearVelocity = Vel
                end
            end
            if MyRoot.Position.Y < -50 then EndFling() end
        end)

        if Mode == "Once" then
            task.delay(1.2, function() if FlingState.Active then EndFling() end end)
        end
    end

        for _, v in pairs(GhostHighlights) do v:Destroy() end

    local function ToggleGhost()
        GhostEnabled = not GhostEnabled
        if GhostEnabled then
            local startCF = Camera.CFrame
            GhostPos = startCF.Position
            local rx, ry, _ = startCF:ToEulerAnglesYXZ()
            GhostRot = Vector2.new(math.deg(rx), math.deg(ry))
            
            if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                LocalPlayer.Character.HumanoidRootPart.Anchored = true
            end

            GhostConn = RunService.BindToRenderStep("FeiGeGhost", Enum.RenderPriority.Camera.Value + 100, function(dt)
                Camera.CameraType = Enum.CameraType.Scriptable
                UserInputService.MouseBehavior = Enum.MouseBehavior.LockCenter
                
                local delta = UserInputService:GetMouseDelta()
                GhostRot = GhostRot - Vector2.new(delta.Y, delta.X) * 0.4
                GhostRot = Vector2.new(math.clamp(GhostRot.X, -89, 89), GhostRot.Y)
                
                local rotCF = CFrame.fromEulerAnglesYXZ(math.rad(GhostRot.X), math.rad(GhostRot.Y), 0)
                local input = Vector3.new()
                if UserInputService:IsKeyDown(Enum.KeyCode.W) then input = input + Vector3.new(0,0,-1) end
                if UserInputService:IsKeyDown(Enum.KeyCode.S) then input = input + Vector3.new(0,0,1) end
                if UserInputService:IsKeyDown(Enum.KeyCode.A) then input = input + Vector3.new(-1,0,0) end
                if UserInputService:IsKeyDown(Enum.KeyCode.D) then input = input + Vector3.new(1,0,0) end
                if UserInputService:IsKeyDown(Enum.KeyCode.E) then input = input + Vector3.new(0,1,0) end
                if UserInputService:IsKeyDown(Enum.KeyCode.Q) then input = input + Vector3.new(0,-1,0) end
                
                local speed = 0.5
                if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then speed = 2 end
                
                if input.Magnitude > 0 then
                    GhostPos = GhostPos + (rotCF:VectorToWorldSpace(input) * speed)
                end
                
                Camera.CFrame = CFrame.new(GhostPos) * rotCF
                
                for _, obj in pairs(Workspace:GetDescendants()) do
                    if obj:IsA("Model") and obj ~= LocalPlayer.Character and (obj:FindFirstChild("Humanoid") or string.find(obj.Name, "Monster")) then
                        if not obj:FindFirstChild("FeiGeGhostHL") then
                            local hl = Instance.new("Highlight")
                            hl.Name = "FeiGeGhostHL"
                            hl.Adornee = obj
                            hl.FillColor = Color3.new(1,0,0)
                            hl.OutlineColor = Color3.new(1,1,1)
                            hl.Parent = CoreGui
                            table.insert(GhostHighlights, hl)
                        end
                    end
                end
            end)
            
            ContextActionService:BindActionAtPriority("FeiGeGhostInput", function() return Enum.ContextActionResult.Sink end, false, 20000, Enum.KeyCode.W, Enum.KeyCode.A, Enum.KeyCode.S, Enum.KeyCode.D, Enum.KeyCode.Space)
            Library:Notify("自由视角已开启", 2)
        else
            if GhostConn then RunService:UnbindFromRenderStep("FeiGeGhost") end
            ContextActionService:UnbindAction("FeiGeGhostInput")
            ClearGhostHighlights()
            if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                LocalPlayer.Character.HumanoidRootPart.Anchored = false
            end
            Camera.CameraType = Enum.CameraType.Custom
            Camera.CameraSubject = LocalPlayer.Character.Humanoid
            UserInputService.MouseBehavior = Enum.MouseBehavior.Default
            Library:Notify("自由视角已关闭", 2)
        end
    end

    RunService.RenderStepped:Connect(function(dt)
        if LocalPlayer.Character then
            if not Toggles.MonitorToggle.Value then
                local hum = LocalPlayer.Character:FindFirstChild("Humanoid")
                local root = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
                if hum and root and hum.MoveDirection.Magnitude > 0 then
                    if Toggles.SpeedToggle.Value then
                        root.AssemblyLinearVelocity = Vector3.new(hum.MoveDirection.X * Options.NormalSlider.Value, root.AssemblyLinearVelocity.Y, hum.MoveDirection.Z * Options.NormalSlider.Value)
                    elseif Toggles.SuperToggle.Value then
                        root.CFrame = root.CFrame + (hum.MoveDirection * (dt * 60 * Options.MultiSlider.Value))
                        root.AssemblyLinearVelocity = Vector3.new(0,0,0)
                    end
                end
            end
        end
        if Toggles.InfJump.Value and game:GetService("UserInputService"):IsKeyDown(Enum.KeyCode.Space) then
            LocalPlayer.Character.Humanoid:ChangeState(3)
        end
        if Toggles.Noclip.Value and LocalPlayer.Character then
            for _,v in pairs(LocalPlayer.Character:GetDescendants()) do if v:IsA("BasePart") then v.CanCollide=false end end
        end
