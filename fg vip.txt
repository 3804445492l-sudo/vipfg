-- 1. 创建 FOV 滑块
local slider = LeftGroupBox:AddSlider("fov", {
    Text = "Field of View", -- 建议改个名区分
    Default = 70,           -- 默认值稍微调大点比较符合通用习惯，可自行改回 20
    Min = 0,
    Max = 120,
    Rounding = 0,
    Suffix = "°",
})

-- 2. [新增] 创建 FOV 锁定开关
local FovLockToggle = LeftGroupBox:AddToggle("FovLock", {
    Text = "锁定 FOV (防修改)",
    Default = false,
    Tooltip = "开启后，游戏系统无法更改你的视野",
})

-- 变量：防止代码修改相机时触发死循环
local updatingFromCode = false
-- 变量：记录当前滑块的值，用于锁定
local currentTargetFov = slider.Value or 70 

-- 3. 当滑块被用户改动时 -> 修改相机
slider:OnChanged(function(value)
    -- 如果是脚本正在同步数据，则不执行
    if updatingFromCode then return end
    
    currentTargetFov = value -- 更新目标值
    
    updatingFromCode = true
    camera.FieldOfView = value
    updatingFromCode = false
end)

-- 4. 当相机 FOV 发生变化时 (核心逻辑)
camera:GetPropertyChangedSignal("FieldOfView"):Connect(function()
    -- 如果是脚本自己改的，忽略
    if updatingFromCode then return end

    -- 检查锁定开关是否开启
    if FovLockToggle.Value then
        -- [锁定模式]
        -- 游戏试图修改 FOV，我们强制把它改回来
        if camera.FieldOfView ~= currentTargetFov then
            updatingFromCode = true
            camera.FieldOfView = currentTargetFov
            updatingFromCode = false
        end
    else
        -- [未锁定模式] (原来的逻辑)
        -- 游戏修改了 FOV，同步更新滑块显示
        updatingFromCode = true
        slider:SetValue(camera.FieldOfView)
        -- 同时更新目标值，防止下次开启锁定时跳变
        currentTargetFov = camera.FieldOfView 
        updatingFromCode = false
    end
end)

-- [额外保护] 如果游戏每一帧都在强制修改FOV，上面的信号可能不够快
-- 只有在开启锁定时，每一帧都强制覆盖 FOV
RunService.RenderStepped:Connect(function()
    if FovLockToggle.Value and not updatingFromCode then
        if camera.FieldOfView ~= currentTargetFov then
            updatingFromCode = true
            camera.FieldOfView = currentTargetFov
            updatingFromCode = false
        end
    end
end)

local MyButton = LeftGroupBox:AddButton({
    Text = "终极透视 (细节加重 + 死亡秒清)",
    Func = function()
        -- // 1. 状态开关 //
        if _G.ComboESP_Active then
            _G.ComboESP_Active = false
            return
        end
        _G.ComboESP_Active = true

        -- // 2. 环境声明 //
        local Players = game:GetService("Players")
        local RunService = game:GetService("RunService")
        local Camera = workspace.CurrentCamera
        local LocalPlayer = Players.LocalPlayer

        -- // 3. 骨骼缓存 //
        local Skeleton_Cache = {}

        -- // 4. 画笔工具 (加重细节) //
        local function NewLine()
            local l = Drawing.new("Line")
            l.Visible = false
            l.Thickness = 2.5 -- 细节加重：加粗线条
            l.Transparency = 1
            l.Color = Color3.new(1, 1, 1)
            l.From = Vector2.new(0, 0)
            l.To = Vector2.new(0, 0)
            return l
        end

        -- // 5. 强效清理函数 //
        local function RemoveESP(plr)
            if Skeleton_Cache[plr] then
                for _, line in pairs(Skeleton_Cache[plr]) do
                    line.Visible = false -- 先隐藏防止闪烁
                    line:Remove()
                end
                Skeleton_Cache[plr] = nil
            end
            if plr.Character then
                local hl = plr.Character:FindFirstChild("SupremeContour")
                if hl then hl:Destroy() end
            end
        end

        -- // 6. 兼容性Part获取 (暴力扫描) //
        local function GetPart(char, names)
            for _, name in pairs(names) do
                local p = char:FindFirstChild(name)
                if p and p:IsA("BasePart") then return p end
            end
            return nil
        end

        -- // 7. 主循环 (RenderStepped 保证丝滑) //
        local function Update()
            if not _G.ComboESP_Active then
                for p, _ in pairs(Skeleton_Cache) do RemoveESP(p) end
                return
            end

            local color = Color3.fromHSV(tick() % 5 / 5, 1, 1)

            for _, player in pairs(Players:GetPlayers()) do
                if player ~= LocalPlayer then
                    local char = player.Character
                    local hum = char and char:FindFirstChildOfClass("Humanoid")

                    -- 【核心修复：严格判断死亡状态】
                    -- 如果没有角色、没有生命值、或者不在地图内，执行 Remove 并跳过
                    if not (char and hum and hum.Health > 0 and char:FindFirstChild("Head")) then
                        RemoveESP(player)
                        continue
                    end

                    -- A. 轮廓强化
                    local hl = char:FindFirstChild("SupremeContour")
                    if not hl then
                        hl = Instance.new("Highlight")
                        hl.Name = "SupremeContour"
                        hl.Parent = char
                        hl.FillTransparency = 1
                        hl.OutlineTransparency = 0
                        hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                    end
                    hl.OutlineColor = color

                    -- B. 骨骼全细节 (R15 全覆盖)
                    if not Skeleton_Cache[player] then
                        Skeleton_Cache[player] = {
                            Head_Upper = NewLine(), Upper_Lower = NewLine(),
                            L_UArm = NewLine(), L_LArm = NewLine(), L_Hand = NewLine(),
                            R_UArm = NewLine(), R_LArm = NewLine(), R_Hand = NewLine(),
                            L_ULeg = NewLine(), L_LLeg = NewLine(), L_Foot = NewLine(),
                            R_ULeg = NewLine(), R_LLeg = NewLine(), R_Foot = NewLine()
                        }
                    end

                    local sk = Skeleton_Cache[player]
                    local function Draw(line, p1, p2)
                        if p1 and p2 then
                            local pos1, vis1 = Camera:WorldToViewportPoint(p1.Position)
                            local pos2, vis2 = Camera:WorldToViewportPoint(p2.Position)
                            if vis1 and vis2 then
                                line.From = Vector2.new(pos1.X, pos1.Y)
                                line.To = Vector2.new(pos2.X, pos2.Y)
                                line.Color = color
                                line.Visible = true
                                return
                            end
                        end
                        line.Visible = false
                    end

                    -- 暴力获取肢体 (适配 R6, R15, 及各类改名 Part)
                    local Head = char:FindFirstChild("Head")
                    local UpperTorso = GetPart(char, {"UpperTorso", "Torso", "HumanoidRootPart"})
                    local LowerTorso = GetPart(char, {"LowerTorso", "Torso", "HumanoidRootPart"})
                    
                    -- 左肢
                    local LUA = GetPart(char, {"LeftUpperArm", "Left Arm"})
                    local LLA = GetPart(char, {"LeftLowerArm", "Left Arm"})
                    local LH  = GetPart(char, {"LeftHand", "Left Arm"})
                    -- 右肢
                    local RUA = GetPart(char, {"RightUpperArm", "Right Arm"})
                    local RLA = GetPart(char, {"RightLowerArm", "Right Arm"})
                    local RH  = GetPart(char, {"RightHand", "Right Arm"})
                    -- 左腿
                    local LUL = GetPart(char, {"LeftUpperLeg", "Left Leg"})
                    local LLL = GetPart(char, {"LeftLowerLeg", "Left Leg"})
                    local LF  = GetPart(char, {"LeftFoot", "Left Leg"})
                    -- 右腿
                    local RUL = GetPart(char, {"RightUpperLeg", "Right Leg"})
                    local RLL = GetPart(char, {"RightLowerLeg", "Right Leg"})
                    local RF  = GetPart(char, {"RightFoot", "Right Leg"})

                    -- 全细节连线
                    Draw(sk.Head_Upper, Head, UpperTorso)
                    Draw(sk.Upper_Lower, UpperTorso, LowerTorso)
                    
                    Draw(sk.L_UArm, UpperTorso, LUA); Draw(sk.L_LArm, LUA, LLA); Draw(sk.L_Hand, LLA, LH)
                    Draw(sk.R_UArm, UpperTorso, RUA); Draw(sk.R_LArm, RUA, RLA); Draw(sk.R_Hand, RLA, RH)
                    
                    Draw(sk.L_ULeg, LowerTorso, LUL); Draw(sk.L_LLeg, LUL, LLL); Draw(sk.L_Foot, LLL, LF)
                    Draw(sk.R_ULeg, LowerTorso, RUL); Draw(sk.R_LLeg, RUL, RLL); Draw(sk.R_Foot, RLL, RF)
                end
            end
        end

        -- // 8. 启动 //
        local conn = RunService.RenderStepped:Connect(Update)

        -- 监控清理 (针对全局变量控制)
        task.spawn(function()
            while _G.ComboESP_Active do task.wait(0.5) end
            conn:Disconnect()
            for _, p in pairs(Players:GetPlayers()) do RemoveESP(p) end
        end)
        
        -- 针对玩家退出的监听
        Players.PlayerRemoving:Connect(RemoveESP)
    end
})

--// 核心功能 //--
local function playRunAnimation(humanoid)
    local animator = humanoid:FindFirstChildWhichIsA("Animator") or Instance.new("Animator", humanoid)
    
    local runAnim = Instance.new("Animation")
    runAnim.AnimationId = runAnimId
    
    runTrack = animator:LoadAnimation(runAnim)
    runTrack.Priority = Enum.AnimationPriority.Action
    runTrack:AdjustSpeed(6) -- 6倍速播放动画
    runTrack:Play()

    -- 只有移动时才播放
    if runConnection then runConnection:Disconnect() end
    runConnection = RunService.RenderStepped:Connect(function()
        if humanoid.MoveDirection.Magnitude > 0 then
            if not runTrack.IsPlaying then 
                runTrack:Play() 
                runTrack:AdjustSpeed(6) 
            end
        else
            if runTrack.IsPlaying then runTrack:Stop() end
        end
    end)
end

function enableEgor()
    local char = LocalPlayer.Character
    if not char then return end
    local hum = char:FindFirstChild("Humanoid")
    if not hum then return end

    hum.WalkSpeed = 3 -- 走路极慢
    playRunAnimation(hum)
end

    if not hum then return end

    hum.WalkSpeed = 16 -- 恢复正常速度

-- 安全回溯 (支持循环逻辑)
function Physics.SafetyReturn(MyRoot, MyHum, CurrentTarget)
    if not State.OriginalCFrame then return end
    
    Gui.Status.Text = "⚠ BRAKING..."
    Gui.Status.TextColor3 = Color3.fromRGB(255, 200, 0)
    
    -- 1. 恢复视角 (如果没在循环，或者循环间隙)
    if State.ViewTarget then
        Camera.CameraSubject = State.ViewTarget.Character and State.ViewTarget.Character:FindFirstChild("Humanoid")
    else
        Camera.CameraSubject = MyHum
    end
    
    -- 2. 物理冻结
    MyRoot.Anchored = true
    
    -- 3. 传送
    MyRoot.CFrame = State.OriginalCFrame + Vector3.new(0, Configuration.ReturnHeight, 0)
    
    -- 4. 刹车
    local BrakeTime = tick()
    local BrakeLoop
    BrakeLoop = RunService.Heartbeat:Connect(function()
        if tick() - BrakeTime > Configuration.SafetyBrakeTime then
            BrakeLoop:Disconnect()
            
            -- 刹车结束
            if MyRoot then 
                MyRoot.Anchored = false 
                MyRoot.AssemblyLinearVelocity = Vector3.zero
                MyRoot.AssemblyAngularVelocity = Vector3.zero
            end
            
            if MyHum then
                MyHum.PlatformStand = false
                MyHum:ChangeState(Enum.HumanoidStateType.GettingUp)
            end
            
            State.IsFlinging = false
            
            -- [循环判断核心]
            if State.LoopTarget and State.LoopTarget == CurrentTarget then
                Gui.Status.Text = "LOOPING: NEXT ATTACK..."
                Gui.Status.TextColor3 = Color3.fromRGB(255, 50, 255)
                task.delay(0.1, function()
                    Physics.ExecuteAttack(State.LoopTarget) -- 递归调用攻击
                end)
            else
                Gui.Status.Text = "SYSTEM READY"
                Gui.Status.TextColor3 = Color3.fromRGB(150, 150, 150)
            end

        else
            -- 持续强制归零
            if MyRoot then
                MyRoot.CFrame = State.OriginalCFrame + Vector3.new(0, Configuration.ReturnHeight, 0)
                MyRoot.AssemblyLinearVelocity = Vector3.zero
                MyRoot.AssemblyAngularVelocity = Vector3.zero
                MyRoot.Velocity = Vector3.zero
            end
        end
    end)
end

-- 攻击逻辑
function Physics.ExecuteAttack(TargetPlayer)
    -- 如果正在攻击且不是为了循环重置，则跳过
    if State.IsFlinging then return end
    if not TargetPlayer or not TargetPlayer.Character then 
        -- 目标丢失，停止循环
        if State.LoopTarget == TargetPlayer then
            State.LoopTarget = nil
            Gui.Status.Text = "TARGET LOST - LOOP STOPPED"
        end
        return 
    end

    local MyChar = LocalPlayer.Character
    local MyRoot = MyChar and MyChar:FindFirstChild("HumanoidRootPart")
    local MyHum = MyChar and MyChar:FindFirstChildOfClass("Humanoid")
    local TargetRoot = TargetPlayer.Character:FindFirstChild("HumanoidRootPart")

    if not MyRoot or not TargetRoot then return end

    State.IsFlinging = true
    -- 只有第一次攻击才记录原点，循环中不更新原点，防止位置偏移
    if not State.OriginalCFrame then
        State.OriginalCFrame = MyRoot.CFrame
    end
    
    Gui.Status.Text = "FLINGING: " .. string.upper(TargetPlayer.DisplayName)
    Gui.Status.TextColor3 = Color3.fromRGB(0, 255, 255)

    -- 强制看目标
    Camera.CameraSubject = TargetRoot

    -- 无碰撞
    local Noclip = RunService.Stepped:Connect(function()
        if MyChar then
            for _, v in pairs(MyChar:GetDescendants()) do
                if v:IsA("BasePart") then v.CanCollide = false end
            end
        end
    end)
    table.insert(State.Connections, Noclip)

    local StartTime = tick()
    local Loop = RunService.Heartbeat:Connect(function()
        if (tick() - StartTime > Configuration.Duration) or not TargetRoot.Parent then
            Physics.Cleanup()
            Physics.SafetyReturn(MyRoot, MyHum, TargetPlayer)
            return
        end

        local TargetVel = TargetRoot.AssemblyLinearVelocity
        local PredictPos = TargetRoot.Position + (TargetVel * 0.1)
        
        MyRoot.AssemblyAngularVelocity = Vector3.new(Configuration.SpinPower, Configuration.SpinPower, Configuration.SpinPower)
        
        -- 垂直震荡
        local Spike = math.sin(tick() * 50) * 1.5
        MyRoot.CFrame = CFrame.new(PredictPos) * CFrame.new(0, Spike, 0)
        
        MyRoot.AssemblyLinearVelocity = Vector3.zero 
    end)
    table.insert(State.Connections, Loop)
end

local MyButton = LeftGroupBox:AddButton({
    Text = "实时翻译 (增强优化版)",
    Func = function()
        --[[
            【死缠烂打·性能优化版 v2.0】
            1. 异步队列：完全不卡 UI，翻译在后台静默排队。
            2. 智能防封：检测到谷歌 API 报错会自动降速。
            3. 递归锁定：防止游戏原生脚本强制改回英文导致的闪烁。
        ]]

        local HttpService = game:GetService("HttpService")
        local Players = game:GetService("Players")
        local Workspace = game:GetService("Workspace")
        local RunService = game:GetService("RunService")

        local CONFIG = {
            TargetLang = "zh-CN",
            Font = Enum.Font.GothamBold,
            Cooldown = 0.1, -- 每个请求的基础间隔
            MaxRetry = 3,   -- 最大重试次数
        }

        local requestFunc = (syn and syn.request) or (http and http.request) or http_request or (fluxus and fluxus.request) or request
        if not requestFunc then return end

        local SuccessCache = {}
        local PendingSet = {}
        local LastRequestTime = 0

        -- === 过滤逻辑：只翻有意义的内容 ===
        local function ShouldTranslate(text, obj)
            if not text or #text < 2 then return false end
            local clean = text:match("^%s*(.-)%s*$")
            if not clean or #clean < 2 or clean == "i" then return false end
            if clean:match("^[%d%p%s]+$") then return false end
            -- 检查是否包含中文（跳过已翻译）
            for i = 1, #clean do if string.byte(clean, i) > 127 then return false end end
            if obj:IsA("TextBox") then return false end
            return true
        end

        local function URL_Encode(str)
            return string.gsub(str, "([^%w %-%_%.%~])", function(c) 
                return string.format("%%%02X", string.byte(c)) 
            end):gsub(" ", "+")
        end

        -- === 核心翻译函数 (异步处理) ===
        local function PerformTranslate(text, callback)
            -- 控制请求频率，防止被封 IP
            local now = tick()
            if now - LastRequestTime < CONFIG.Cooldown then
                task.wait(CONFIG.Cooldown - (now - LastRequestTime))
            end
            LastRequestTime = tick()

            local url = "https://translate.googleapis.com/translate_a/single?client=gtx&sl=auto&tl="..CONFIG.TargetLang.."&dt=t&q="..URL_Encode(text)
            
            task.spawn(function()
                local success, resp = pcall(function() return requestFunc({Url = url, Method = "GET"}) end)
                
                if success and resp.StatusCode == 200 then
                    local data = HttpService:JSONDecode(resp.Body)
                    if data and data[1] then
                        local result = ""
                        for _, line in ipairs(data[1]) do result = result .. (line[1] or "") end
                        callback(true, result)
                    else
                        callback(false)
                    end
                elseif resp and resp.StatusCode == 429 then
                    warn("Google API 请求过快，正在进入冷静期...")
                    task.wait(5) -- 触发限流，强制休眠
                    callback(false)
                else
                    callback(false)
                end
            end)
        end

        -- === 智能分配任务 ===
        local function AddToWork(obj)
            local originalText = obj.Text
            if not ShouldTranslate(originalText, obj) then return end
            
            -- 如果有缓存，直接秒改
            if SuccessCache[originalText] then
                obj.Text = SuccessCache[originalText]
                obj.Font = CONFIG.Font
                return
            end

            -- 防止重复进入翻译队列
            if PendingSet[originalText] then return end
            PendingSet[originalText] = true

            task.spawn(function()
                local solved = false
                local retry = 0
                
                while not solved and retry < CONFIG.MaxRetry do
                    local done = false
                    PerformTranslate(originalText, function(ok, res)
                        if ok and res then
                            SuccessCache[originalText] = res
                            if obj and obj.Parent then
                                obj.Text = res
                                obj.Font = CONFIG.Font
                            end
                            solved = true
                        end
                        done = true
                    end)
                    
                    -- 等待该次异步请求结束
                    local start = tick()
                    repeat task.wait() until done or (tick() - start > 5)
                    
                    if not solved then
                        retry = retry + 1
                        task.wait(retry * 1) -- 失败后递增重试间隔
                    end
                end
                PendingSet[originalText] = nil
            end)
        end

        -- === 监听与挂钩 ===
        local function Hook(obj)
            if not (obj:IsA("TextLabel") or obj:IsA("TextButton")) then return end
            
            AddToWork(obj)
            
            -- 监控文字变化
            obj:GetPropertyChangedSignal("Text"):Connect(function()
                local currentText = obj.Text
                -- 如果新文字不在缓存里，且不是我们刚刚翻好的文字，才去翻译
                if not SuccessCache[currentText] and ShouldTranslate(currentText, obj) then
                    AddToWork(obj)
                end
            end)
        end

        -- === 扫描初始化 ===
        local function DeepScan(root)
            for _, v in ipairs(root:GetDescendants()) do Hook(v) end
            root.DescendantAdded:Connect(function(v)
                task.wait(0.1) -- 等待属性加载
                Hook(v)
            end)
        end

        -- 扫全图
        task.spawn(function() DeepScan(Players.LocalPlayer:WaitForChild("PlayerGui")) end)
        task.spawn(function()
            for _, v in ipairs(Workspace:GetDescendants()) do
                if v:IsA("BillboardGui") or v:IsA("SurfaceGui") then DeepScan(v) end
            end
        end)

        -- === 扫地僧巡逻 (防止漏网之鱼和闪烁) ===
        task.spawn(function()
            while true do
                for _, obj in ipairs(Players.LocalPlayer.PlayerGui:GetDescendants()) do
                    if (obj:IsA("TextLabel") or obj:IsA("TextButton")) and obj.Visible then
                        local t = obj.Text
                        if SuccessCache[t] then
                            -- 修正被游戏脚本改回来的情况
                            -- 暂不处理，防止部分动态数值显示异常
                        elseif ShouldTranslate(t, obj) and not PendingSet[t] then
                            AddToWork(obj)
                        end
                    end
                end
                task.wait(4) -- 巡逻频率调低至4秒，保证性能
            end
        end)

        print(">>> 增强版翻译机启动：异步队列模式。")
    end,
    -- 下方保持你的按钮配置不变...
})

local MOVE_SMOOTHNESS = 0.08
local LOOK_SMOOTHNESS = 0.2

-- 高亮设置 (鲜红色，方便看清纸片人)
local ESP_COLOR = Color3.new(1, 0, 0)   -- 红色
local ESP_FILL_TRANSPARENCY = 0.5       -- 填充透明度
local ESP_OUTLINE_TRANSPARENCY = 0      -- 轮廓不透明 (最亮)

--=== 变量 ===--
local Enabled = false
local CurrentPos = Vector3.new()
local CurrentVelocity = Vector3.new()
local CurrentRot = Vector2.new()
local TargetRot = Vector2.new()
local Highlights = {} -- 存高亮物体

local RENDER_NAME = "SuperGhostCam_ESP"
local ACTION_NAME = "GhostInputSink"

--=== 🛡️ 辅助功能：高亮透视 ===--
local function AddHighlight(model)
    if model == LocalPlayer.Character then return end -- 不高亮自己
    if model:FindFirstChild("Highlight_Ghost") then return end -- 已经有了

    -- 创建高亮
    local hl = Instance.new("Highlight")
    hl.Name = "Highlight_Ghost"
    hl.FillColor = ESP_COLOR
    hl.OutlineColor = Color3.new(1, 1, 1) -- 白色轮廓
    hl.FillTransparency = ESP_FILL_TRANSPARENCY
    hl.OutlineTransparency = ESP_OUTLINE_TRANSPARENCY
    hl.Adornee = model
    hl.Parent = CoreGui -- 放在CoreGui防止被游戏清理
    
    table.insert(Highlights, hl)
end

local function ScanEnemies()
    -- 扫描所有模型，寻找可能是敌人的东西
    for _, obj in pairs(Workspace:GetDescendants()) do
        -- 如果是有Humanoid的(大部分NPC/怪物)，或者名字里带"Cat"/"Kitty"的
        if (obj:IsA("Model") and obj:FindFirstChild("Humanoid")) or 
           (obj:IsA("Model") and (string.find(obj.Name, "Cat") or string.find(obj.Name, "Kitty") or string.find(obj.Name, "Monster"))) then
            AddHighlight(obj)
        end
    end
end

local function ClearHighlights()
    for _, hl in pairs(Highlights) do
        if hl then hl:Destroy() end
    end
    Highlights = {}
end

--=== 🎥 核心逻辑 ===--

local function MakeCharVisible()
    local char = LocalPlayer.Character
    if char then
        for _, part in pairs(char:GetDescendants()) do
            if part:IsA("BasePart") then
                part.LocalTransparencyModifier = 0
            end
        end
    end
end

local function SinkInputs(actionName, inputState, inputObject)
    return Enum.ContextActionResult.Sink
end

local function UpdateCam(dt)
    if not Enabled then return end

    -- 1. 暴力维持摄像机状态
    if Camera.CameraType ~= Enum.CameraType.Scriptable then
        Camera.CameraType = Enum.CameraType.Scriptable
    end
    UserInputService.MouseBehavior = Enum.MouseBehavior.LockCenter
    UserInputService.MouseIconEnabled = false

    -- 2. 鼠标输入
    local delta = UserInputService:GetMouseDelta()
    TargetRot = TargetRot - Vector2.new(delta.Y, delta.X) * SENSITIVITY
    TargetRot = Vector2.new(math.clamp(TargetRot.X, -89, 89), TargetRot.Y)

    -- 3. 平滑计算
    local timeScale = math.clamp(dt * 60, 0, 5)
    CurrentRot = CurrentRot:Lerp(TargetRot, LOOK_SMOOTHNESS * timeScale)
    local newRotationCF = CFrame.fromEulerAnglesYXZ(math.rad(CurrentRot.X), math.rad(CurrentRot.Y), 0)

    -- 4. 移动
    local inputDir = Vector3.new()
    if UserInputService:IsKeyDown(Enum.KeyCode.W) then inputDir = inputDir + Vector3.new(0, 0, -1) end
    if UserInputService:IsKeyDown(Enum.KeyCode.S) then inputDir = inputDir + Vector3.new(0, 0, 1) end
    if UserInputService:IsKeyDown(Enum.KeyCode.A) then inputDir = inputDir + Vector3.new(-1, 0, 0) end
    if UserInputService:IsKeyDown(Enum.KeyCode.D) then inputDir = inputDir + Vector3.new(1, 0, 0) end

    local speed = BASE_SPEED
    if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then speed = BASE_SPEED * SHIFT_MULTIPLIER end
    if UserInputService:IsKeyDown(Enum.KeyCode.LeftAlt) then speed = BASE_SPEED * 0.5 end

    local targetVel = Vector3.new()
    if inputDir.Magnitude > 0 then
        targetVel = (newRotationCF:VectorToWorldSpace(inputDir)).Unit * speed
    end
    
    CurrentVelocity = CurrentVelocity:Lerp(targetVel, MOVE_SMOOTHNESS * timeScale)
    CurrentPos = CurrentPos + (CurrentVelocity * timeScale)

    Camera.CFrame = CFrame.new(CurrentPos) * newRotationCF
    MakeCharVisible()
end

--=== 🚀 开关 ===--
local function Toggle()
    Enabled = not Enabled
    
    if Enabled then
        -- 开启
        local startCF = Camera.CFrame
        CurrentPos = startCF.Position
        local rx, ry, rz = startCF:ToEulerAnglesYXZ()
        CurrentRot = Vector2.new(math.deg(rx), math.deg(ry))
        TargetRot = CurrentRot
        CurrentVelocity = Vector3.new()

        -- 锁人
        local char = LocalPlayer.Character
        if char and char:FindFirstChild("HumanoidRootPart") then
            char.HumanoidRootPart.Anchored = true
        end
        
        -- 开启扫描 (高亮敌人)
        ScanEnemies()

        -- 绑定渲染 (最高优先级)
        RunService:BindToRenderStep(RENDER_NAME, Enum.RenderPriority.Last.Value, UpdateCam)

        -- 屏蔽输入
        ContextActionService:BindActionAtPriority(
            ACTION_NAME, SinkInputs, false, 10000,
            Enum.KeyCode.W, Enum.KeyCode.A, Enum.KeyCode.S, Enum.KeyCode.D,
            Enum.KeyCode.Space, Enum.KeyCode.LeftShift, Enum.KeyCode.Tab,
            Enum.PlayerActions.CharacterJump
        )
        
        StarterGui:SetCore("SendNotification", {Title="透视幽灵模式", Text="已开启 (敌人已高亮)", Duration=2})
    else
        -- 关闭
        RunService:UnbindFromRenderStep(RENDER_NAME)
        ContextActionService:UnbindAction(ACTION_NAME)
        ClearHighlights() -- 清除高亮
        
        local char = LocalPlayer.Character
        if char and char:FindFirstChild("HumanoidRootPart") then
            char.HumanoidRootPart.Anchored = false
        end
        
        Camera.CameraType = Enum.CameraType.Custom
        UserInputService.MouseBehavior = Enum.MouseBehavior.Default
        UserInputService.MouseIconEnabled = true
        
        StarterGui:SetCore("SendNotification", {Title="模式关闭", Text="回到身体", Duration=1})
    end
end

--=== 2. 互换传送 (X) ===--
local function SwapTeleport()
    local char = LocalPlayer.Character
    if not char then return end
    local root = char:FindFirstChild("HumanoidRootPart")
    if not root then return end

    if not ActiveDot or not ActiveDot.Parent then
        StarterGui:SetCore("SendNotification", {Title="提示", Text="❌ 请先按 Z 标记坐标", Duration=1})
        return
    end

    -- 刹车函数
    local function StopPhysics()
        root.Velocity = Vector3.new(0,0,0)
        root.AssemblyLinearVelocity = Vector3.new(0,0,0)
        root.AssemblyAngularVelocity = Vector3.new(0,0,0)
    end

    -- 计算你离标记点有多远
    local dist = (root.Position - ActiveDot.Position).Magnitude

    -- 播放音效
    local s = Instance.new("Sound", root)
    s.SoundId = "rbxassetid://1146749005"
    s:Play()
    Debris:AddItem(s, 1)

    -- === 核心判断 ===
    if dist > DETECT_RANGE then
        -- [情况A：你在外面] -> 飞回家
        
        -- 1. 记下你在哪，方便等下返回
        SavedReturnPos = root.CFrame
        
        -- 2. 传送到标记点
        root.CFrame = ActiveDot.CFrame + SAFE_HEIGHT
        StopPhysics()
        
        StarterGui:SetCore("SendNotification", {Title="飞雷神", Text="🏠 回到标记点", Duration=0.5})
        
    else
        -- [情况B：你在标记点附近] -> 返回原来的地方
        
        if SavedReturnPos then
            -- 1. 传回刚才记录的点
            root.CFrame = SavedReturnPos
            StopPhysics()
            
            -- 这里不清空记录，允许你反复横跳
            StarterGui:SetCore("SendNotification", {Title="飞雷神", Text="↩️ 返回原位", Duration=0.5})
        else
            StarterGui:SetCore("SendNotification", {Title="提示", Text="⚠️ 没有记录返回点", Duration=0.5})
        end
    end
end

-- // 3. 动作提取函数 (核心修改) //
local function GetRunAnimation()
    local char = LocalPlayer.Character
    if not char then return nil end
    local hum = char:FindFirstChild("Humanoid")
    local animate = char:FindFirstChild("Animate")
    
    if hum and animate then
        -- 尝试找到 Roblox 自带的跑步动作 ID
        -- 兼容 R6 和 R15
        local runVal = animate:FindFirstChild("run") or animate:FindFirstChild("walk")
        if runVal then
            local animObj = runVal:FindFirstChildWhichIsA("Animation")
            if animObj then
                local animator = hum:FindFirstChildOfClass("Animator") or hum
                return animator:LoadAnimation(animObj)
            end
        end
    end
    return nil
end

-- // 4. 循环逻辑 //
Services.RunService.Heartbeat:Connect(function(deltaTime)
    if not Enabled then return end
    
    local char = LocalPlayer.Character
    if not char then return end
    local hum = char:FindFirstChild("Humanoid")
    local root = char:FindFirstChild("HumanoidRootPart")
    
    if hum and root then
        -- 如果在移动 (按了WASD)
        if hum.MoveDirection.Magnitude > 0 then
            
            -- [A] 暴力位移 (TP)
            root.CFrame = root.CFrame + (hum.MoveDirection * (deltaTime * 50 * SpeedMultiplier))
            
            -- [B] 防止滑步 (仅保留Y轴速度以允许跳跃)
            root.AssemblyLinearVelocity = Vector3.new(0, root.AssemblyLinearVelocity.Y, 0)
            
            -- [C] 动作强制播放逻辑
            -- 1. 检测是否在地面 (空中不需要播放跑步动作)
            local ray = Ray.new(root.Position, Vector3.new(0, -4, 0))
            local hitPart, _ = workspace:FindPartOnRayWithIgnoreList(ray, {char})
            
            if hitPart then
                -- 在地面，准备播放动作
                if not RunAnimationTrack then
                    RunAnimationTrack = GetRunAnimation() -- 获取动作
                end
                
                if RunAnimationTrack and not RunAnimationTrack.IsPlaying then
                    RunAnimationTrack:Play() -- 强制播放
                end
                
                -- 根据你的TP速度调整动作播放速度，看起来更自然
                if RunAnimationTrack then
                    RunAnimationTrack:AdjustSpeed(SpeedMultiplier * 0.8)
                end
            else
                -- 在空中 (跳跃中)，停止跑步动作，让系统播放跳跃动作
                if RunAnimationTrack and RunAnimationTrack.IsPlaying then
                    RunAnimationTrack:Stop()
                end
            end
            
        else
            -- 没按键盘，停止动作
            if RunAnimationTrack and RunAnimationTrack.IsPlaying then
                RunAnimationTrack:Stop()
            end
        end
    end
end)

local lastChar = nil
local bv, bp, bg = nil, nil, nil

RunService.RenderStepped:Connect(function(dt)
    local char = player.Character
    if not char or not char.Parent then
        -- 等待重生
        lastChar = nil
        return
    end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local humanoid = char:FindFirstChildWhichIsA("Humanoid")
    if not hrp or not humanoid then return end

    if enabled then
        if char ~= lastChar then
            -- 角色变化时（重生），确保控制器存在
            destroyControllers(hrp)
            bv, bp, bg = createControllers(hrp)
            lastChar = char
        end

        -- 锁高度逻辑：保持 targetY，按 E 上升，Q 下降（相对平滑）
        if keys.E then
            targetY = targetY + ascendSpeed * dt
        elseif keys.Q then
            targetY = targetY - ascendSpeed * dt
        end
        if targetY == nil then targetY = hrp.Position.Y end

        -- 计算移动方向（相对于摄像头朝向）
        local camera = workspace.CurrentCamera
        local look = camera.CFrame.LookVector
        local right = camera.CFrame.RightVector
        -- 仅取平面分量并规范化
        look = Vector3.new(look.X, 0, look.Z).Unit
        right = Vector3.new(right.X, 0, right.Z).Unit

        local mv = Vector3.new(0,0,0)
        if keys.W then mv = mv + look end
        if keys.S then mv = mv - look end
        if keys.A then mv = mv - right end
        if keys.D then mv = mv + right end
        if mv.Magnitude > 0 then
            mv = mv.Unit
        end

        -- 设定 BodyVelocity 的水平速度
        if bv and bv.Parent then
            bv.Velocity = Vector3.new(mv.X * speed, 0, mv.Z * speed)
        end

        -- 设定 BodyPosition 的 Y（高度保持）并跟随 XZ 到 HRP 当前位置（避免身体被甩开）
        if bp and bp.Parent then
            -- 保持 XZ 跟随当前 HRP 位置（让水平移动由 BodyVelocity 控制）
            bp.Position = Vector3.new(hrp.Position.X, targetY, hrp.Position.Z)
        end

        -- BodyGyro 让身体保持朝向（随摄像头朝向）
        if bg and bg.Parent then
            bg.CFrame = CFrame.new(hrp.Position, hrp.Position + Vector3.new(look.X, 0, look.Z))
        end

        -- 禁用 Humanoid 的默认物理影响（避免与 Body* 冲突）
        humanoid.PlatformStand = true
    else
        -- 如果未启用，确保清理（在角色存在时）
        if lastChar == char then
            destroyControllers(hrp)
            if humanoid then humanoid.PlatformStand = false end
            lastChar = nil
            bv, bp, bg = nil, nil, nil
        end
    end
end)
